async function generate() { const input = document.getElementById("input"); const topic = input.value.trim(); if (!topic) return; addMsg(topic, "user"); saveHistory(topic); input.value = ""; const loading = addMsg("Thinking", "ai"); loading.classList.add("dots"); try { const res = await fetch("/generate", { method: "POST", headers: {"Content-Type":"application/json"}, body: JSON.stringify({ topic }) }); const data = await res.json(); const pitch = data.idea.pitch; const output = `🚀 STARTUP IDEA ${data.idea.startup_idea} ❗ PROBLEM ${data.idea.problem} 💡 SOLUTION ${data.idea.solution} 📊 PITCH One-liner: ${pitch.one_liner} Problem: ${pitch.problem_statement} Solution: ${pitch.solution_statement} Market: ${pitch.market} Business Model: ${pitch.business_model} Why Now: ${pitch.why_now} MVP: - ${pitch.mvp.join("\n- ")} 💰 Monetization: ${data.idea.monetization}`; loading.classList.remove("dots"); typeText(loading, output); } catch (e) { loading.innerText = "Backend error ❌"; } }
